1.3 3D wireframe plane

  gr()
  
  x1=range(-4,4,length=11)
    y1=range(-5,5,length=11)
    f(x1,y1) = -x1/6+y1/2
    
    # either the function is supplied directly, or
    # it is applied to the ranges, with the following
    # transposed x, and output suppressed with ;
    
    z1 = f.(x1',y1);
    
    wireframe!(x1,y1,z1,)
    scatter!([0],[0],[0],marker=:circle)
