1.3 Histograms

using Distributions, DataFramesMeta
    
default(legend=false,
        size=(400,400))
    
iris1 = @rsubset(iris, :Species=="setosa")
iris2 = @rsubset(iris, :Species=="versicolor")
iris3 = @rsubset(iris, :Species=="virginica")
    
p1 = @df iris1 histogram(:SepalLength, color=:lightblue, title="setosa", bins=6)
p2 = @df iris2 histogram(:SepalLength, color=:lightgreen, title="versicolor", bins=8)
p3 = @df iris3 histogram(:SepalLength, color=:orange, title="virginica", bins=8)    

plot(p1,p2,p3, layout=(2,2), xlim=[3,9], ylim=[0,20])
