These plots are based on parametric functions.

direction: right
a: |latex t |
b: |latex  x=f(t) |
c: |latex  y=g(t) |
d: |latex  plot(x,y) |

1 Parametrics

1.1 Polynomials

# parametric parabola
t = range(-2,1, length=20) |> collect
x = @. t^2+t
y = @. 2*t- 1

plot(x,y, color="red", framestyle=:origin, 
       leg=false, size=(300,250), lw=2)

1.2 Trigonometric

# ellipse
t = range(0,2*pi, length=100)
x = 5*cos.(t)
y = 2*sin.(t)
plot(x,y, framestyle=:origin, leg=false,
      size=(300,250) , lw=2)
# Rose Curve
t = range(0,2*pi, length=200)
x = @. cos(t)*sin(4*t)
y = @. sin(t)*sin(4*t)
plot(x,y, framestyle=:origin, leg=false,
       size=(300,250),lw=2)
# Lissajous
t = range(-12,12, length=600)
x = @. 2*cos(3*t)
y = @. 2*sin(2*t)
plot(x,y, leg=false,
      size=(300,300),lw=2,linecolor=:red)
# Hypotrochoid
t = range(-12,12, length=600)
x = @. 2*cos(t)+ 5*cos(2*t/3)
y = @. 2*sin(t)- 5*sin(2*t/3)
plot(x,y, leg=false,
      size=(300,300), lw=2)
# Hypotrochoid with variables
R=15
r=14
d=1

t = range(0,90, length=800)
x = @. (R-r)*cos(t)+ d*cos(((R-r)/r)*t)
y = @. (R-r)*sin(t)- d*sin(((R-r)/r)*t)
plot(x,y , leg=false,
      size=(300,300), lw=2, linecolor=:brown)
